 94 /**
 95 * @scope enchant.PhysicsWorld.prototype
 96 */
 97 enchant.PhysicsWorld = enchant.Class.create({
 98     /**
 99     * 物理シミュレーションを行う世界のクラス
100     * @example
101     *   //y軸方向へ重力加速度9.8m/s^2
102     *   var physicsWorld = new PhysicsWorld(0, 9.8);
103     *   //無重力
104     *   var physicsWorld = new PhysicsWorld(0, 0);
105     * 
106     * @param {Number} [gravityX] x軸方向への引力.
107     * @param {Number} [gravityY] y軸方向への引力.
108     * @constructs
109     */
110     initialize: function (gravityX, gravityY) {
111         var game = enchant.Game.instance;
112         this.jointShowSurface = new Surface(game.width, game.height);
113         /**
114         * 物理シミュレーションの精度
115         * @type {Nunber}
116         */
117         this.iterations = 10;
118         world = new b2World(
119             new b2Vec2(gravityX, gravityY)  //gravity
120             , true                          //allow sleep
121         );
122     },
123     /**
124     * 物理シミュレーション内の時間を進める
125     * @param {b2Vec2} [pos] Spriteの座標.
126     */
127     step: function (fps) {
128         world.Step(1 / fps, this.iterations, this.iterations);
129     },
130     /**
131     * 物体の当たり判定
132     * @example
133     *   //ぶつかった2つのSpriteを消す
134     *   physicsWorld.contact(function (sprite1, sprite2) {
135     *       sprite1.destroy();
136     *       sprite2.destroy(); 
137     *   });
138     *
139     * @param {function(sprite1:enchant.PhySprite,sprite2:enchant.PhySprite)} [func] 当たり判定時の処理
140     */
141     contact: function (func) {
142         var c = world.m_contactList;
143         if (c) {
144             for (var contact = c; contact; contact = contact.m_next) {
145                 var pos1 = contact.m_fixtureA.m_body.GetPosition().Copy();
146                 pos1.Subtract(contact.m_fixtureB.m_body.GetPosition());
147                 pos1.Multiply(WORLD_SCALE);
148                 var r1 = (contact.m_fixtureA.m_body.m_userData.width + contact.m_fixtureB.m_body.m_userData.width) / 2;
149                 var r2 = (contact.m_fixtureA.m_body.m_userData.height + contact.m_fixtureB.m_body.m_userData.height) / 2;
150                 if (Math.abs(pos1.x) <= r1 && Math.abs(pos1.y) <= r2) {
151                     func(contact.m_fixtureA.m_body.m_userData, contact.m_fixtureB.m_body.m_userData);
152                 }
153             }
154         }
155     },
156     getJointImage: function () {
157         var surface = this.jointShowSurface;
158         surface.context.clearRect(0, 0, surface.width, surface.height);
159         var drawJoint = function (joint) {
160             var b1 = joint.m_bodyA;
161             var b2 = joint.m_bodyB;
162             var x1 = b1.GetPosition().Copy();
163             var x2 = b2.GetPosition().Copy();
164             var p1 = joint.GetAnchorA().Copy();
165             var p2 = joint.GetAnchorB().Copy();
166             x1.Multiply(WORLD_SCALE);
167             x2.Multiply(WORLD_SCALE);
168             p1.Multiply(WORLD_SCALE);
169             p2.Multiply(WORLD_SCALE);
170             surface.context.strokeStyle = DrawJointColor;
171             surface.context.beginPath();
172             switch (joint.m_type) {
173                 case Box2D.Dynamics.Joints.b2Joint.e_distanceJoint:
174                     surface.context.moveTo(p1.x, p1.y);
175                     surface.context.lineTo(p2.x, p2.y);
176                     break;
177 
178                 case Box2D.Dynamics.Joints.b2Joint.e_pulleyJoint:
179                     var gu1 = joint.GetGroundAnchorA().Copy();
180                     var gu2 = joint.GetGroundAnchorB().Copy();
181                     gu1.Multiply(WORLD_SCALE);
182                     gu2.Multiply(WORLD_SCALE);
183                     surface.context.moveTo(x1.x, x1.y);
184                     surface.context.lineTo(p1.x, p1.y);
185                     surface.context.lineTo(gu1.x, gu1.y);
186                     surface.context.lineTo(gu2.x, gu2.y);
187                     surface.context.lineTo(x2.x, x2.y);
188                     surface.context.lineTo(p2.x, p2.y);
189                     break;
190 
191                 default:
192                     if (b1 == world.m_groundBody) {
193                         surface.context.moveTo(p1.x, p1.y);
194                         surface.context.lineTo(x2.x, x2.y);
195                     }
196                     else if (b2 == world.m_groundBody) {
197                         surface.context.moveTo(p1.x, p1.y);
198                         surface.context.lineTo(x1.x, x1.y);
199                     }
200                     else {
201                         surface.context.moveTo(x1.x, x1.y);
202                         surface.context.lineTo(p1.x, p1.y);
203                         surface.context.lineTo(x2.x, x2.y);
204                         surface.context.lineTo(p2.x, p2.y);
205                     }
206                     break;
207             }
208             surface.context.stroke();
209         }
210         for (var j = world.m_jointList; j; j = j.m_next) {
211             drawJoint(j);
212         }
213         return surface;
214     },
215     /**
216     * ワールドにあるすべてのオブジェクトを削除
217     * シーンの切り替わり時に呼んでおかないと、次のシーンでも衝突判定がおこってしまう。
218     */
219     cleanAllSprite: function () {
220         var body = world.m_bodyList;
221         var nextBody;
222         while (body) {
223             nextBody = body.m_next;
224             if (body.m_userData) {
225                 body.m_userData.destroy();
226             }
227             else {
228                 world.DestroyBody(body);
229             }
230             body = nextBody;
231         }
232     },
233     /**
234     * ワールドにあるすべてのジョイントを削除
235     */
236     cleanAllJoint: function () {
237         var joint = world.m_jointList;
238         var nextJoint;
239         while (joint) {
240             nextJoint = joint.m_next;
241             if (joint.m_userData) {
242                 joint.m_userData.destroy();
243             }
244             else {
245                 world.DestroyJoint(joint);
246             }
247             joint = nextJoint;
248         }
249     },
250     /**
251     * ワールドのすべてを削除
252     */
253     cleanUp: function () {
254         this.cleanAllJoint();
255         this.cleanAllSprite();
256     }
257 });

260 /**
261 * @scope enchant.PhySprite.prototype
262 */
263 enchant.PhySprite = enchant.Class.create(enchant.Sprite, {
264     /**
265     * 画像表示機能を持った物理シミュレーションクラス.
266     * @param {Number} [width] Spriteの横幅.
267     * @param {Number} [height] Spriteの高さ.
268     * @constructs
269     * @extends enchant.Sprite
270     */
271     initialize: function (width, height) {
272         this.body;
273         enchant.Sprite.call(this, width, height);
274 
275         var time = 0;
276         this.addEventListener(enchant.Event.ENTER_FRAME, function (e) {
277             this.x = this.x;
278             this.y = this.y;
279             if (time % 2) {   //なぜか移動と回転を一緒にできない。謎。
280                 this.rotation = this.angle;
281             }
282             time = (time + 1) % 2;
283         });
284     },
285     /**
286     * 四角形の物理シミュレーション用Sprite生成.
287     * @param {Boolean} type 静的,動的,キネマティック
288     * @param {Number} density Spriteの密度.
289     * @param {Number} friction Spriteの摩擦.
290     * @param {Number} restitution Spriteの反発.
291     * @param {Boolean} isSleeping Spriteが初めから物理演算を行うか.
292     */
293     createPhyBox: function (type, density, friction, restitution, awake) {
294         var fixDef = new b2FixtureDef;
295         fixDef.density = (density != null ? density : 1.0);             // 密度
296         fixDef.friction = (friction != null ? friction : 0.5);          // 摩擦
297         fixDef.restitution = (restitution != null ? restitution : 0.3); // 反発
298         fixDef.shape = new b2PolygonShape;
299         fixDef.shape.SetAsBox(this.width / 2 / WORLD_SCALE, this.height / 2 / WORLD_SCALE);
300         var bodyDef = new b2BodyDef;
301         bodyDef.type = type;
302         bodyDef.position.x = 0;
303         bodyDef.position.y = 0;
304         bodyDef.awake = (awake != null ? awake : true);
305         bodyDef.userData = this;
306         this.body = world.CreateBody(bodyDef).CreateFixture(fixDef);
307         this.setDegugImage();
308         return this.body;
309     },
310     /**
311     * 多角形の物理シミュレーション用Sprite生成.
312     * @param {b2Vec2[]} vertexs 多角形の頂点配列
313     * @param {Boolean} type 静的,動的,キネマティック
314     * @param {Number} density Spriteの密度.
315     * @param {Number} friction Spriteの摩擦.
316     * @param {Number} restitution Spriteの反発.
317     * @param {Boolean} isSleeping Spriteが初めから物理演算を行うか.
318     */
319     createPhyPolygon: function (vertexs, type, density, friction, restitution, awake) {
320         for (var i = 0; i < vertexs.length; i++) {
321             vertexs[i].Multiply(1 / WORLD_SCALE);
322         }
323         var fixDef = new b2FixtureDef;
324         fixDef.density = (density != null ? density : 1.0);             // 密度
325         fixDef.friction = (friction != null ? friction : 0.5);          // 摩擦
326         fixDef.restitution = (restitution != null ? restitution : 0.3); // 反発
327         fixDef.shape = new b2PolygonShape;
328         fixDef.shape.SetAsArray(vertexs, vertexs.length);
329         var bodyDef = new b2BodyDef;
330         bodyDef.type = type;
331         bodyDef.position.x = 0;
332         bodyDef.position.y = 0;
333         bodyDef.awake = (awake != null ? awake : true);
334         bodyDef.userData = this;
335         this.body = world.CreateBody(bodyDef).CreateFixture(fixDef);
336         this.setDegugImage();
337         return this.body;
338     },
339     /**
340     * 円形の物理シミュレーション用Sprite生成.
341     * @param {Boolean} type 静的,動的,キネマティック
342     * @param {Number} density Spriteの密度.
343     * @param {Number} friction Spriteの摩擦.
344     * @param {Number} restitution Spriteの反発.
345     * @param {Boolean} isSleeping Spriteが初めから物理演算を行うか.
346     */
347     createPhyCircle: function (type, density, friction, restitution, awake) {
348         var fixDef = new b2FixtureDef;
349         fixDef.density = (density != null ? density : 1.0);             // 密度
350         fixDef.friction = (friction != null ? friction : 0.5);          // 摩擦
351         fixDef.restitution = (restitution != null ? restitution : 0.3); // 反発
352         fixDef.shape = new b2CircleShape(this.width / 2 / WORLD_SCALE);
353         var bodyDef = new b2BodyDef;
354         bodyDef.type = type;
355         bodyDef.position.x = 0;
356         bodyDef.position.y = 0;
357         bodyDef.awake = (awake != null ? awake : true);
358         bodyDef.userData = this;
359         this.body = world.CreateBody(bodyDef).CreateFixture(fixDef);
360         this.setDegugImage();
361         return this.body;
362     },
363     /**
364     * デバッグ用のイメージをセット
365     */
366     setDegugImage: function () {
367         var surface = new Surface(this.width, this.height);
368         surface.context.strokeStyle = DebugDrawStrokeColor;
369         surface.context.fillStyle = DebugDrawFillColor;
370         surface.context.beginPath();
371         var shape = this.body.GetShape();
372         switch (shape.m_type) {
373             case b2Shape.e_circleShape:
374                 {
375                     var circle = shape;
376                     var r = circle.m_radius * WORLD_SCALE - 1;
377                     surface.context.arc(this.width / 2, this.height / 2, r, 0, Math.PI * 2, true);
378                     surface.context.moveTo(this.width / 2, this.height / 2);
379                     surface.context.lineTo(this.width - 1, this.height / 2);
380                 }
381                 break;
382             case b2Shape.e_polygonShape:
383                 {
384                     var poly = shape;
385                     var tV = poly.m_vertices[0];
386                     surface.context.moveTo(tV.x * WORLD_SCALE + this.width / 2, tV.y * WORLD_SCALE + this.height / 2);
387                     for (var i = 0; i < poly.m_vertexCount; i++) {
388                         var v = poly.m_vertices[i];
389                         surface.context.lineTo(v.x * WORLD_SCALE + this.width / 2, v.y * WORLD_SCALE + this.height / 2);
390                     }
391                     surface.context.lineTo(tV.x * WORLD_SCALE + this.width / 2, tV.y * WORLD_SCALE + this.height / 2);
392                 }
393                 break;
394         }
395         surface.context.fill();
396         surface.context.stroke();
397         this.image = surface;
398         return surface;
399     },
400     /**
401     * Spriteのタイプ 静的（STATIC_SPRITE）か動的（DYNAMIC_SPRITE)かキネマティック(KINEMATIC_SPRITE)か
402     * @type {bool}
403     */
404     type: {
405         get: function () {
406             return this.body.m_body.GetType();
407         },
408         set: function (type) {
409             this.body.m_body.SetType(type);
410         }
411     },
412     /**
413     * Spriteのx座標.
414     * @type {Number}
415     */
416     x: {
417         get: function () {
418             return this.body.m_body.GetPosition().x * WORLD_SCALE - this.width / 2;
419         },
420         set: function (x) {
421             this._x = x;
422             x += this.width / 2;
423             this.body.m_body.SetPosition(new b2Vec2(x / WORLD_SCALE, this.body.m_body.GetPosition().y));
424             this._updateCoordinate();
425         }
426     },
427     /**
428     * Spriteのy座標.
429     * @type {Number}
430     */
431     y: {
432         get: function () {
433             return this.body.m_body.GetPosition().y * WORLD_SCALE - this.height / 2;
434         },
435         set: function (y) {
436             this._y = y;
437             y += this.height / 2;
438             this.body.m_body.SetPosition(new b2Vec2(this.body.m_body.GetPosition().x, y / WORLD_SCALE));
439             this._updateCoordinate();
440         }
441     },
442     /**
443     * Spriteの中心のx座標.
444     * @type {Number}
445     */
446     centerX: {
447         get: function () {
448             return this.x + this.width / 2;
449         },
450         set: function (x) {
451             this.x = x - this.width / 2;
452         }
453     },
454     /**
455     * Spriteの中心のy座標.
456     * @type {Number}
457     */
458     centerY: {
459         get: function () {
460             return this.y + this.height / 2;
461         },
462         set: function (y) {
463             this.y = y - this.height / 2;
464         }
465     },
466     /**
467     * Spriteの中心座標ベクトル.
468     * @type {b2Vec2}
469     */
470     position: {
471         get: function () {
472             var pos = this.body.m_body.GetPosition().Copy();
473             pos.Multiply(WORLD_SCALE);
474             return pos;
475         },
476         set: function (pos) {
477             this.centerX = pos.x;
478             this.centerY = pos.y;
479             this.body.m_body.SetPosition(new b2Vec2(pos.x / WORLD_SCALE, pos.y / WORLD_SCALE));
480         }
481     },
482     /**
483     * Spriteのx座標の速度（単位はpx/s）.
484     * @type {Number}
485     */
486     vx: {
487         get: function () {
488             return this.body.m_body.GetLinearVelocity().x * WORLD_SCALE;
489         },
490         set: function (x) {
491             this.body.m_body.SetLinearVelocity(new b2Vec2(x / WORLD_SCALE, this.body.m_body.GetLinearVelocity().y));
492         }
493     },
494     /**
495     * Spriteのy座標の速度（単位はpx/s）.
496     * @type {Number}
497     */
498     vy: {
499         get: function () {
500             return this.body.m_body.GetLinearVelocity().y * WORLD_SCALE;
501         },
502         set: function (y) {
503             this.body.m_body.SetLinearVelocity(new b2Vec2(this.body.m_body.GetLinearVelocity().x, y / WORLD_SCALE));
504         }
505     },
506     /**
507     * Spriteの速度（単位はpx/s）.
508     * @type {b2Vec2}
509     */
510     velocity: {
511         get: function () {
512             var v = this.body.m_body.GetLinearVelocity().Copy();
513             v.Multiply(WORLD_SCALE);
514             return v;
515         },
516         set: function (v) {
517             this.body.m_body.SetLinearVelocity(new b2Vec2(v.x / WORLD_SCALE, v.y / WORLD_SCALE));
518         }
519     },
520     /**
521     * Spriteの角度 (度数法)..
522     * @type {Number}
523     */
524     angle: {
525         get: function () {
526             return this.body.m_body.GetAngle() * (180 / Math.PI);
527         },
528         set: function (angle) {
529             this.rotation = angle;
530             this.body.m_body.SetAngle(angle * (Math.PI / 180));
531         }
532     },
533     /**
534     * Spriteの角速度（単位はdeg/s）.
535     * @type {b2Vec2}
536     */
537     angularVelocity: {
538         get: function () {
539             return this.body.m_body.GetAngularVelocity() * (180 / Math.PI);
540         },
541         set: function (omega) {
542             this.setAwake(true);
543             this.body.m_body.SetAngularVelocity(omega * (Math.PI / 180));
544         }
545     },
546     /**
547     * 継続的な力を加える
548     * @param {b2Vec2} force 加える力のベクトル
549     */
550     applyForce: function (force) {
551         this.setAwake(true);
552         this.body.m_body.ApplyForce(force, this.body.m_body.GetPosition());
553     },
554     /**
555     * 瞬間的な力を加える
556     * @param {b2Vec2} impulse 加える力のベクトル
557     */
558     applyImpulse: function (impulse) {
559         this.setAwake(true);
560         this.body.m_body.ApplyImpulse(impulse, this.body.m_body.GetPosition());
561     },
562     /**
563     * 継続的な回転力を与える
564     * @param {Number} torque 加える回転力
565     */
566     applyTorque: function (torque) {
567         this.setAwake(true);
568         this.body.m_body.ApplyTorque(torque);
569     },
570     /**
571     * 物理シミュレーションされているか
572     * 物体の動きが止まると処理コスト軽減のためsleep状態になる
573     * @type {Boolean}
574     */
575     sleep: {
576         get: function () {
577             return this.body.m_body.IsSleepingAllowed();
578         },
579         set: function (flag) {
580             this.setAwake(true);
581             this.body.m_body.SetSleepingAllowed(flag);
582         }
583     },
584     /**
585     * 物理シミュレーションされていない時、物理シミュレーションを行う(sleep時は動かなくなるので)
586     * @param {Boolean} flag 物理シミュレーションを行うかどうか
587     */
588     setAwake: function (flag) {
589         this.body.m_body.SetAwake(flag);
590     },
591     /**
592     * アクティブかどうか
593     * 物理シミュレーションが行われて、他の物体に干渉できるか
594     * @type {Boolean}
595     */
596     active: {
597         get: function () {
598             return this.body.m_body.IsActive();
599         },
600         set: function (flag) {
601             this.body.m_body.SetActive(flag);
602         }
603     },
604     /**
605     * 表示/非表示(物理シミュレーションも止まる)
606     * @type {Boolean}
607     */
608     visible: {
609         get: function () {
610             return this.active;
611         },
612         set: function (visible) {
613             if (this._visible = visible) {
614                 this._style.display = 'block';
615             } else {
616                 this._style.display = 'none';
617             }
618             this.active = visible;
619         }
620     },
621     /**
622     * 静的オブジェクトかどうか
623     * @return {Boolean}
624     */
625     isStatic: function () {
626         return (this.type == STATIC_SPRITE);
627     },
628     /**
629     * 動的オブジェクトかどうか
630     * @return {Boolean}
631     */
632     isDynamic: function () {
633         return (this.type == DYNAMIC_SPRITE);
634     },
635     /**
636     * キネマティックオブジェクトかどうか
637     * @return {Boolean}
638     */
639     isKinematic: function () {
640         return (this.type == KINEMATIC_SPRITE);
641     },
642     /**
643     * 衝突判定
644     * @example
645     *   //bearに当たったSpriteを消す
646     *   bear.contact(function (sprite) {
647     *      sprite.destroy(); 
648     *   });
649     * 
650     * @param {function(sprite:enchant.PhySprite)} [func] ぶつかったSpriteを引数とする関数
651     */
652     contact: function (func) {
653         var c = this.body.m_body.m_contactList;
654         if (c) {
655             for (var contact = c.contact; contact; contact = contact.m_next) {
656                 var pos1 = contact.m_fixtureA.m_body.GetPosition().Copy();
657                 pos1.Subtract(contact.m_fixtureB.m_body.GetPosition());
658                 pos1.Multiply(WORLD_SCALE);
659                 var r1 = (contact.m_fixtureA.m_body.m_userData.width + contact.m_fixtureB.m_body.m_userData.width) / 1.5;
660                 var r2 = (contact.m_fixtureA.m_body.m_userData.height + contact.m_fixtureB.m_body.m_userData.height) / 1.5;
661                 if (Math.abs(pos1.x) <= r1 && Math.abs(pos1.y) <= r2) {
662                     //片方が自分ならもう片方をぶつかった相手として処理する
663                     if (this.body.m_body == contact.m_fixtureA.m_body)
664                         func(contact.m_fixtureB.m_body.m_userData);
665                     else if (this.body.m_body == contact.m_fixtureB.m_body)
666                         func(contact.m_fixtureA.m_body.m_userData);
667                 }
668             }
669         }
670     },
671     /**
672     * 物体の削除
673     * removeChildではなくこちらでSpriteを取り除く
674     */
675     destroy: function () {
676         if (this.scene != null) {
677             world.DestroyBody(this.body.m_body);
678             this.body.Destroy();
679             this.scene.removeChild(this);
680         }
681     },
682     /**
683     * bodyの取得
684     * Box2Dの処理を各自行いたい時に取得する
685     */
686     getBody: function () {
687         return this.body;
688     }
