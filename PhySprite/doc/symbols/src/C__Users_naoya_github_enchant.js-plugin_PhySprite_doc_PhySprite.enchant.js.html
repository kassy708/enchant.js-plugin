 13 /**
 14 * Spriteの種類（スタティック）
 15 * @type {Number}
 16 */
 17 var STATIC_SPRITE = 0;
 18 /**
 19 * Spriteの種類（ダイナミック）
 20 * @type {Number}
 21 */
 22 var DYNAMIC_SPRITE = 1;

 25 var WORLD_SCALE = 32;
 26 var world;

 44 /**
 45 * @scope enchant.PhysicsWorld.prototype
 46 */
 47 enchant.PhysicsWorld = enchant.Class.create({
 48     /**
 49     * 物理シミュレーションを行う世界のクラス
 50     * @example
 51     *   //y軸方向へ重力加速度9.8m/s^2
 52     *   var physicsWorld = new PhysicsWorld(0, 9.8);
 53     *   //無重力
 54     *   var physicsWorld = new PhysicsWorld(0, 0);
 55     * 
 56     * @param {Number} [gravityX] x軸方向への引力.
 57     * @param {Number} [gravityY] y軸方向への引力.
 58     * @constructs
 59     */
 60     initialize: function (gravityX, gravityY) {
 61         /**
 62         * 物理シミュレーションの精度
 63         * @type {Nunber}
 64         */
 65         this.iterations = 10;    
 66         world = new b2World(
 67             new b2Vec2(gravityX, gravityY)  //gravity
 68             , true                          //allow sleep
 69         );
 70     },
 71     /**
 72     * 物理シミュレーション内の時間を進める
 73     * @param {b2Vec2} [pos] Spriteの座標.
 74     */
 75     step: function (fps) {
 76         world.Step(1 / fps, this.iterations, this.iterations);
 77     },
 78     /**
 79     * 物体の当たり判定
 80     * @example
 81     *   //ぶつかった2つのSpriteを消す
 82     *   physicsWorld.contact(function (sprite1, sprite2) {
 83     *       sprite1.destroy();
 84     *       sprite2.destroy(); 
 85     *   });
 86     *
 87     * @param {function(sprite1:enchant.PhySprite,sprite2:enchant.PhySprite)} [func] 当たり判定時の処理
 88     */
 89     contact: function (func) {
 90         var c = world.m_contactList;
 91         if (c) {
 92             for (var contact = c; contact; contact = contact.m_next) {
 93                 func(contact.m_fixtureA.m_body.m_userData, contact.m_fixtureB.m_body.m_userData);
 94             }
 95         }
 96     }
 97 });

100 /**
101 * @scope enchant.PhySprite.prototype
102 */
103 enchant.PhySprite = enchant.Class.create(enchant.Sprite, {
104     /**
105     * 画像表示機能を持った物理シミュレーションクラス.
106     * @param {Number} [width] Spriteの横幅.
107     * @param {Number} [height] Spriteの高さ.
108     * @constructs
109     * @extends enchant.Sprite
110     */
111     initialize: function (width, height) {
112         this.body;
113         this.staticOrDynamic;
114         enchant.Sprite.call(this, width, height);
115 
116         var time = 0;
117         this.addEventListener(enchant.Event.ENTER_FRAME, function (e) {
118             if (this.staticOrDynamic == DYNAMIC_SPRITE) {
119                 var pos = this.position;
120                 this.x = pos.x - this.width / 2;
121                 this.y = pos.y - this.height / 2;
122                 if (time % 2) {   //なぜか座標と回転を一緒にできない。謎。
123                     this.x = pos.x - this.width / 2;
124                     this.y = pos.y - this.height / 2;
125                 }
126                 else
127                     this.rotation = this.angle;
128                 time++;
129             }
130         });
131     },
132     /**
133     * 四角形の物理シミュレーション用Sprite生成.
134     * @param {Boolean} staticOrDynamic 静止するか動くか.
135     * @param {Number} density Spriteの密度.
136     * @param {Number} friction Spriteの摩擦.
137     * @param {Number} restitution Spriteの反発.
138     * @param {Boolean} isSleeping Spriteが初めから物理演算を行うか.
139     */
140     createPhyBox: function (staticOrDynamic, density, friction, restitution, awake) {
141         this.staticOrDynamic = staticOrDynamic;
142         var fixDef = new b2FixtureDef;
143         fixDef.density = (density != null ? density : 1.0);             // 密度
144         fixDef.friction = (friction != null ? friction : 0.5);          // 摩擦
145         fixDef.restitution = (restitution != null ? restitution : 0.3); // 反発
146         fixDef.shape = new b2PolygonShape;
147         fixDef.shape.SetAsBox(this.width / 2 / WORLD_SCALE, this.height / 2 / WORLD_SCALE);
148         var bodyDef = new b2BodyDef;
149         bodyDef.type = (staticOrDynamic == STATIC_SPRITE ? b2Body.b2_staticBody : b2Body.b2_dynamicBody);
150         bodyDef.position.x = 0;
151         bodyDef.position.y = 0;
152         bodyDef.awake = (awake != null ? awake : true);
153         bodyDef.userData = this;
154         return world.CreateBody(bodyDef).CreateFixture(fixDef);
155     },
156     /**
157     * 円形の物理シミュレーション用Sprite生成.
158     * @param {Boolean} staticOrDynamic 静止するか動くか.
159     * @param {Number} density Spriteの密度.
160     * @param {Number} friction Spriteの摩擦.
161     * @param {Number} restitution Spriteの反発.
162     * @param {Boolean} isSleeping Spriteが初めから物理演算を行うか.
163     */
164     createPhyCircle: function (staticOrDynamic, density, friction, restitution, awake) {
165         this.staticOrDynamic = staticOrDynamic;
166         var fixDef = new b2FixtureDef;
167         fixDef.density = (density != null ? density : 1.0);             // 密度
168         fixDef.friction = (friction != null ? friction : 0.5);          // 摩擦
169         fixDef.restitution = (restitution != null ? restitution : 0.3); // 反発
170         fixDef.shape = new b2CircleShape(this.width / 2 / WORLD_SCALE);
171         var bodyDef = new b2BodyDef;
172         bodyDef.type = (staticOrDynamic == STATIC_SPRITE ? b2Body.b2_staticBody : b2Body.b2_dynamicBody);
173         bodyDef.position.x = 0;
174         bodyDef.position.y = 0;
175         bodyDef.awake = (awake != null ? awake : true);
176         bodyDef.userData = this;
177         return world.CreateBody(bodyDef).CreateFixture(fixDef);
178     },
179     /**
180     * Spriteの座標.
181     * @type {b2Vec2}
182     */
183     position: {
184         get: function () {
185             var pos = this.body.m_body.GetPosition().Copy();
186             pos.Multiply(WORLD_SCALE);
187             return pos;
188         },
189         set: function (pos) {
190             this.x = pos.x - this.width / 2;
191             this.y = pos.y - this.height / 2;
192             this.body.m_body.SetPosition(new b2Vec2(pos.x / WORLD_SCALE, pos.y / WORLD_SCALE));
193         }
194     },
195     /**
196     * Spriteの速度（単位はpx/s）.
197     * @type {b2Vec2}
198     */
199     velocity: {
200         get: function () {
201             var v = this.body.m_body.GetLinearVelocity().Copy();
202             v.Multiply(WORLD_SCALE);
203             return v;
204         },
205         set: function (v) {
206             this.body.m_body.SetLinearVelocity(new b2Vec2(v.x / WORLD_SCALE, v.y / WORLD_SCALE));
207         }
208     },
209     /**
210     * Spriteの角度 (度数法)..
211     * @type {Number}
212     */
213     angle: {
214         get: function () {
215             return this.body.m_body.GetAngle() * (180 / Math.PI);
216         },
217         set: function (angle) {
218             this.rotation = angle;
219             this.body.m_body.SetAngle(angle * (Math.PI / 180));
220         }
221     },
222     /**
223     * Spriteの角速度（単位はdeg/s）.
224     * @type {b2Vec2}
225     */
226     angularVelocity: {
227         get: function () {
228             return this.body.m_body.GetAngularVelocity() * (180 / Math.PI);
229         },
230         set: function (omega) {
231             this.setAwake(true);
232             this.body.m_body.SetAngularVelocity(omega * (Math.PI / 180));
233         }
234     },
235     /**
236     * 継続的な力を加える
237     * @param {b2Vec2} force 加える力のベクトル
238     */
239     applyForce: function (force) {
240         this.setAwake(true);
241         this.body.m_body.ApplyForce(force, this.body.m_body.GetPosition());
242     },
243     /**
244     * 瞬間的な力を加える
245     * @param {b2Vec2} impulse 加える力のベクトル
246     */
247     applyImpulse: function (impulse) {
248         this.setAwake(true);
249         this.body.m_body.ApplyImpulse(impulse, this.body.m_body.GetPosition());
250     },
251     /**
252     * 継続的な回転力を与える
253     * @param {Number} torque 加える回転力
254     */
255     applyTorque: function (torque) {
256         this.setAwake(true);
257         this.body.m_body.ApplyTorque(torque);
258     },
259     /**
260     * 物理シミュレーションされているか
261     * @type {Boolean}
262     */
263     sleep: {
264         get: function () {
265             return this.body.m_body.IsSleepingAllowed();
266         },
267         set: function (flag) {
268             this.setAwake(true);
269             this.body.m_body.SetSleepingAllowed(flag);
270         }
271     },
272     /**
273     * 物理シミュレーションされていない時、物理シミュレーションを行う(sleep時は動かなくなるので)
274     * @param {Boolean} flag 物理シミュレーションを行うかどうか
275     */
276     setAwake: function (flag) {
277         this.body.m_body.SetAwake(flag);
278     },
279     /**
280     * 衝突判定(当たり判定でかい？)
281     * @example
282     *   //bearに当たったSpriteを消す
283     *   bear.contact(function (sprite) {
284     *      sprite.destroy(); 
285     *   });
286     * 
287     * @param {function(sprite:enchant.PhySprite)} [func] ぶつかったSpriteを引数とする関数
288     */
289     contact: function (func) {
290         var c = this.body.m_body.m_contactList;
291         if (c) {
292             for (; c; c = c.m_next) {
293                 for (var contact = c.contact; contact; contact = contact.m_next) {
294                     //片方が自分ならもう片方をぶつかった相手として処理する
295                     if (this.body.m_body == contact.m_fixtureA.m_body)
296                         func(contact.m_fixtureB.m_body.m_userData);
297                     if (this.body.m_body == contact.m_fixtureB.m_body)
298                         func(contact.m_fixtureA.m_body.m_userData);
299                 }
300             }
301         }
302     },
303     /**
304     * 物体の削除
305     * removeChildではなくこちらでSpriteを取り除く
306     */
307     destroy: function () {
308         if (this.scene != null) {
309             world.DestroyBody(this.body.m_body);
310             this.scene.removeChild(this);
311         }
312     }
313 
314 });

319 enchant.PhyBoxSprite = enchant.Class.create(enchant.PhySprite, {
320     /**
321     * 四角形の物理シミュレーション用Sprite
322     * @example
323     *   var bear = new PhyBoxSprite(32, 32, DYNAMIC_SPRITE, 1.0, 0.5, 0.3, true);
324     *   bear.image = game.assets['chara1.gif'];
325     * 
326     * @param {Number} [width] Spriteの横幅.
327     * @param {Number} [height] Spriteの高さ.
328     * @param {Boolean}   [staticOrDynamic] 静止するか動くか.
329     * @param {Number} [density] Spriteの密度.
330     * @param {Number} [friction] Spriteの摩擦.
331     * @param {Number} [restitution] Spriteの反発.
332     * @param {Boolean}   [isSleeping] Spriteが初めから物理演算を行うか.
333     * @constructs
334     * @extends enchant.PhySprite
335     */
336     initialize: function (width, height, staticOrDynamic, density, friction, restitution, isSleeping) {
337         enchant.PhySprite.call(this, width, height);
338 
339         //物理オブジェクトの生成
340         this.body = this.createPhyBox(staticOrDynamic, density, friction, restitution, isSleeping);
341     }

365         enchant.PhySprite.call(this, radius * 2, radius * 2);
366 
367         //物理オブジェクトの生成

369     }
